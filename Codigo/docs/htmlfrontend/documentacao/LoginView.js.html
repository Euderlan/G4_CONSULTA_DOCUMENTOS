/**

import React, { useState } from 'react';
import { MessageSquare, User, } from 'lucide-react';
import GoogleLoginButton from '../GoogleLoginButton/GoogleLoginButton';
import './LoginView.css';

/**
 * Componente de interface de login e registro
 * 
 * @component
 * @param {Object} props - Propriedades do componente
 * @param {string} props.API_BASE_URL - URL base da API do backend
 * @param {Function} props.onLoginSuccess - Callback executado quando login √© bem-sucedido
 * @param {boolean} props.isLoading - Estado de carregamento da autentica√ß√£o
 * @param {Function} props.setIsLoading - Fun√ß√£o para alterar estado de carregamento
 * @returns {JSX.Element} Componente renderizado da interface de login
 * 
 * @example
 * <LoginView
 *   API_BASE_URL="http://localhost:8000"
 *   onLoginSuccess={(user, token) => console.log('Login:', user)}
 *   isLoading={false}
 *   setIsLoading={setLoading}
 * />
 */
const LoginView = ({ 
  API_BASE_URL,
  onLoginSuccess,
  isLoading,
  setIsLoading
}) => {
  // === ESTADOS LOCAIS DO COMPONENTE ===
  
  // Estado para credenciais de login tradicional
  /** @type {Object} Credenciais de login */
  const [credentials, setCredentials] = useState({ 
    email: '', 
    password: '' 
  });
  
  // Estado para dados de registro de novo usu√°rio
  /** @type {Object} Dados para registro de novo usu√°rio */
  const [registerData, setRegisterData] = useState({ 
    name: '', 
    email: '', 
    password: '', 
    confirmPassword: '' 
  });
  
  // Estado para controlar qual formul√°rio est√° ativo (login ou registro)
  /** @type {string} Modo atual ('traditional' ou 'register') */
  const [loginMode, setLoginMode] = useState('traditional');

  // === FUN√á√ïES DE VALIDA√á√ÉO ===
  
  /**
   * Valida formato de email usando express√£o regular
   * @function validateEmail
   * @param {string} email - Email a ser validado
   * @returns {boolean} True se email √© v√°lido, false caso contr√°rio
   * 
   * @example
   * validateEmail("user@ufma.br") // true
   * validateEmail("invalid-email") // false
   */
  const validateEmail = (email) => {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  };

  /**
   * Valida formul√°rio de login antes do envio
   * @function validateLoginForm
   * @returns {boolean} True se formul√°rio √© v√°lido, false caso contr√°rio
   */
  const validateLoginForm = () => {
    if (!credentials.email || !credentials.password) {
      alert('Por favor, preencha email e senha!');
      return false;
    }

    if (!validateEmail(credentials.email)) {
      alert('Por favor, insira um email v√°lido!');
      return false;
    }

    return true;
  };

  /**
   * Valida formul√°rio de registro com m√∫ltiplas verifica√ß√µes
   * @function validateRegisterForm
   * @returns {boolean} True se formul√°rio √© v√°lido, false caso contr√°rio
   */
  const validateRegisterForm = () => {
    // Valida√ß√£o de campos obrigat√≥rios
    if (!registerData.name || !registerData.email || !registerData.password) {
      alert('Preencha todos os campos!');
      return false;
    }
    
    // Valida√ß√£o de nome
    if (registerData.name.trim().length < 2) {
      alert('O nome deve ter pelo menos 2 caracteres!');
      return false;
    }
    
    // Valida√ß√£o de email
    if (!validateEmail(registerData.email)) {
      alert('Por favor, insira um email v√°lido!');
      return false;
    }
    
    // Valida√ß√£o de senha
    if (registerData.password.length < 6) {
      alert('A senha deve ter pelo menos 6 caracteres!');
      return false;
    }
    
    // Valida√ß√£o de confirma√ß√£o de senha
    if (registerData.password !== registerData.confirmPassword) {
      alert('As senhas n√£o coincidem!');
      return false;
    }

    return true;
  };

  // === FUN√á√ÉO DE LOGIN TRADICIONAL ===
  
  /**
   * Processa login com email e senha via API
   * @async
   * @function handleLogin
   * @returns {Promise<void>}
   * @throws {Error} Erro de conex√£o ou autentica√ß√£o
   */
  const handleLogin = async () => {
    if (!validateLoginForm()) return;

    setIsLoading(true);
    try {
      const response = await fetch(`${API_BASE_URL}/api/login`, {
        method: 'POST',
        headers: { 
          'Content-Type': 'application/x-www-form-urlencoded' 
        },
        body: new URLSearchParams({ 
          username: credentials.email, 
          password: credentials.password 
        }).toString(),
      });
      
      const data = await response.json();
      
      if (response.ok) {
        // Sucesso - chama callback do App.js para atualizar estado global
        onLoginSuccess(data.user, data.access_token);
        
        // Limpa formul√°rio ap√≥s sucesso
        setCredentials({ email: '', password: '' });
      } else {
        // Erro - mostra mensagem de erro espec√≠fica
        alert('Erro no login: ' + (data.detail || 'Email ou senha incorretos.'));
      }
    } catch (error) {
      console.error('Erro no login:', error);
      alert('Erro de conex√£o. Verifique sua internet e tente novamente.');
    } finally {
      setIsLoading(false);
    }
  };

  // === FUN√á√ÉO DE REGISTRO ===
  
  /**
   * Processa cria√ß√£o de nova conta de usu√°rio
   * @async
   * @function handleRegister
   * @returns {Promise<void>}
   * @throws {Error} Erro de conex√£o ou registro
   */
  const handleRegister = async () => {
    if (!validateRegisterForm()) return;

    setIsLoading(true);
    try {
      const response = await fetch(`${API_BASE_URL}/api/login/register`, {
        method: 'POST',
        headers: { 
          'Content-Type': 'application/json' 
        },
        body: JSON.stringify({
          name: registerData.name.trim(),
          email: registerData.email.toLowerCase(),
          password: registerData.password
        }),
      });
      
      const data = await response.json();
      
      if (response.ok) {
        // Sucesso - mostra mensagem e redireciona para login
        alert('üéâ Cadastro realizado com sucesso!\n\nAgora voc√™ pode fazer login com suas credenciais.');
        
        // Limpa formul√°rio de registro
        setRegisterData({ 
          name: '', 
          email: '', 
          password: '', 
          confirmPassword: '' 
        });
        
        // Volta para tela de login
        setLoginMode('traditional');
        
        // Pr√©-preenche email no formul√°rio de login
        setCredentials({ 
          email: registerData.email, 
          password: '' 
        });
      } else {
        // Erro - mostra mensagem espec√≠fica do backend
        const errorMessage = data.detail || 'Erro desconhecido no cadastro.';
        alert('Erro no cadastro: ' + errorMessage);
      }
    } catch (error) {
      console.error('Erro no cadastro:', error);
      alert('Erro de conex√£o. Verifique sua internet e tente novamente.');
    } finally {
      setIsLoading(false);
    }
  };

  // === FUN√á√ïES DO GOOGLE LOGIN ===
  
  /**
   * Processa login bem-sucedido via Google OAuth
   * @async
   * @function handleGoogleSuccess
   * @param {Object} userData - Dados do usu√°rio do Google
   * @param {string} userData.google_token - Token JWT do Google
   * @param {string} userData.email - Email do usu√°rio
   * @param {string} userData.name - Nome do usu√°rio
   * @returns {Promise<void>}
   */
  const handleGoogleSuccess = async (userData) => {
    setIsLoading(true);
    try {
      console.log('Google Login Success:', userData);
      
      const response = await fetch(`${API_BASE_URL}/api/login/google`, {
        method: 'POST',
        headers: { 
          'Content-Type': 'application/json' 
        },
        body: JSON.stringify(userData),
      });
      
      const data = await response.json();
      
      if (response.ok) {
        // Sucesso - chama callback do App.js
        onLoginSuccess(data.user, data.access_token);
      } else {
        alert('Erro no login com Google: ' + (data.detail || 'Erro desconhecido.'));
      }
    } catch (error) {
      console.error('Erro no login com Google:', error);
      alert('Erro de conex√£o com Google. Tente novamente.');
    } finally {
      setIsLoading(false);
    }
  };

  /**
   * Trata erros do Google OAuth
   * @function handleGoogleError
   * @param {Error} error - Erro ocorrido durante autentica√ß√£o Google
   * @returns {void}
   */
  const handleGoogleError = (error) => {
    console.error('Google Login Error:', error);
    alert('Falha no login com Google. Tente novamente.');
  };

  // === FUN√á√ÉO PARA TESTE R√ÅPIDO (DESENVOLVIMENTO) ===
  
  /**
   * Preenche credenciais rapidamente para testes de desenvolvimento
   * @function handleQuickLogin
   * @param {string} email - Email para preenchimento autom√°tico
   * @param {string} password - Senha para preenchimento autom√°tico
   * @returns {void}
   */
  const handleQuickLogin = (email, password) => {
    setCredentials({ email, password });
    // Pequeno delay para mostrar que preencheu
    setTimeout(() => {
      handleLogin();
    }, 100);
  };

  // === HANDLERS DE FORMUL√ÅRIO ===
  
  /**
   * Permite envio de formul√°rio com tecla Enter
   * @function handleKeyDown
   * @param {KeyboardEvent} e - Evento de teclado
   * @param {Function} action - Fun√ß√£o a ser executada quando Enter for pressionado
   * @returns {void}
   */
  const handleKeyDown = (e, action) => {
    if (e.key === 'Enter' && !isLoading) {
      e.preventDefault();
      action();
    }
  };

  /**
   * Gerencia troca entre abas de login e registro
   * @function handleTabSwitch
   * @param {string} mode - Modo a ser ativado ('traditional' ou 'register')
   * @returns {void}
   */
  const handleTabSwitch = (mode) => {
    setLoginMode(mode);
    // Limpa formul√°rios ao trocar de aba para evitar confus√£o
    if (mode === 'traditional') {
      setRegisterData({ name: '', email: '', password: '', confirmPassword: '' });
    } else {
      setCredentials({ email: '', password: '' });
    }
  };

  // === RENDER ===
  return (
    <div className="login-container">
      {/* Anima√ß√£o de fundo decorativa */}
      <div className="background-animation">
        <div className="floating-element element-1"></div>
        <div className="floating-element element-2"></div>
        <div className="floating-element element-3"></div>
        <div className="floating-element element-4"></div>
      </div>

      <div className="login-card">
        {/* Cabe√ßalho com branding do sistema */}
        <div className="login-header">
          <div className="logo-container">
            <MessageSquare size={40} />
          </div>
          <h1 className="system-title">ConsultAI</h1>
          <p className="system-subtitle">Consultas Inteligentes de Documentos</p>
          <p className="document-version">Resolu√ß√µes da UFMA</p>
        </div>

        {/* Conte√∫do principal com formul√°rios */}
        <div className="login-content">
          {/* Bot√µes de navega√ß√£o entre Login e Cadastro */}
          <div className="tab-buttons">
            <button
              onClick={() => handleTabSwitch('traditional')}
              className={`tab-button ${loginMode === 'traditional' ? 'active-tab-blue' : 'inactive-tab'}`}
              disabled={isLoading}
            >
              Login
            </button>
            <button
              onClick={() => handleTabSwitch('register')}
              className={`tab-button ${loginMode === 'register' ? 'active-tab-green' : 'inactive-tab'}`}
              disabled={isLoading}
            >
              Cadastro
            </button>
          </div>

          {/* Formul√°rio de Login */}
          {loginMode === 'traditional' ? (
            <div className="form-container">
              <div className="form-group">
                <label className="form-label">Email</label>
                <input
                  type="email"
                  className="form-input"
                  placeholder="Digite seu email"
                  value={credentials.email}
                  onChange={(e) => setCredentials({...credentials, email: e.target.value})}
                  onKeyDown={(e) => handleKeyDown(e, handleLogin)}
                  disabled={isLoading}
                />
              </div>
              
              <div className="form-group">
                <label className="form-label">Senha</label>
                <input
                  type="password"
                  className="form-input"
                  placeholder="Digite sua senha"
                  value={credentials.password}
                  onChange={(e) => setCredentials({...credentials, password: e.target.value})}
                  onKeyDown={(e) => handleKeyDown(e, handleLogin)}
                  disabled={isLoading}
                />
              </div>
              
              <div className="button-group">
                {/* Bot√£o principal de login */}
                <button
                  onClick={handleLogin}
                  disabled={isLoading}
                  className="primary-button"
                >
                  {isLoading ? (
                    <div className="loading-content">
                      <div className="spinner"></div>
                      Entrando...
                    </div>
                  ) : (
                    <>
                      <User className="button-icon" />
                      Entrar
                    </>
                  )}
                </button>
                
                {/* Componente de login com Google */}
                <GoogleLoginButton
                  clientId={process.env.REACT_APP_GOOGLE_CLIENT_ID}
                  onSuccess={handleGoogleSuccess}
                  onError={handleGoogleError}
                  disabled={isLoading}
                  buttonText="Entrar com Google"
                />
                
                {/* Se√ß√£o de teste r√°pido com credenciais pr√©-definidas */}
                <div>
                  <p className="info-text"></p>
                  <div className="admin-credentials">
                    <div>
                      <span 
                        onClick={() => !isLoading && handleQuickLogin('admin@ufma.br', 'admin123')}
                        style={{ 
                          cursor: isLoading ? 'not-allowed' : 'pointer',
                          textDecoration: 'underline',
                          opacity: isLoading ? 0.5 : 1
                        }}
                      ></span>
                    </div>
                    <div>
                      <span 
                        onClick={() => !isLoading && handleQuickLogin('usuario@gmai.com', 'user123')}
                        style={{ 
                          cursor: isLoading ? 'not-allowed' : 'pointer',
                          textDecoration: 'underline',
                          opacity: isLoading ? 0.5 : 1
                        }}
                      ></span>
                    </div>
                  </div>
                </div>
              </div>
            </div>
          ) : (
            /* Formul√°rio de Registro */
            <div className="form-container">
              <div className="form-group">
                <label className="form-label">Nome Completo</label>
                <input
                  type="text"
                  className="form-input-green"
                  placeholder="Digite seu nome completo"
                  value={registerData.name}
                  onChange={(e) => setRegisterData({...registerData, name: e.target.value})}
                  disabled={isLoading}
                />
              </div>
              
              <div className="form-group">
                <label className="form-label">Email</label>
                <input
                  type="email"
                  className="form-input-green"
                  placeholder="Digite seu email"
                  value={registerData.email}
                  onChange={(e) => setRegisterData({...registerData, email: e.target.value})}
                  disabled={isLoading}
                />
              </div>
              
              <div className="form-group">
                <label className="form-label">Senha</label>
                <input
                  type="password"
                  className="form-input-green"
                  placeholder="Digite sua senha (m√≠n. 6 caracteres)"
                  value={registerData.password}
                  onChange={(e) => setRegisterData({...registerData, password: e.target.value})}
                  disabled={isLoading}
                />
              </div>
              
              <div className="form-group">
                <label className="form-label">Confirmar Senha</label>
                <input
                  type="password"
                  className="form-input-green"
                  placeholder="Confirme sua senha"
                  value={registerData.confirmPassword}
                  onChange={(e) => setRegisterData({...registerData, confirmPassword: e.target.value})}
                  onKeyDown={(e) => handleKeyDown(e, handleRegister)}
                  disabled={isLoading}
                />
              </div>
              
              {/* Bot√£o de cria√ß√£o de conta */}
              <button
                onClick={handleRegister}
                disabled={isLoading}
                className="register-button"
              >
                {isLoading ? (
                  <div className="loading-content">
                    <div className="spinner"></div>
                    Criando conta...
                  </div>
                ) : (
                  <>
                    <User className="button-icon" />
                    Criar Conta
                  </>
                )}
              </button>
            </div>
          )}
        </div>

        {/* Rodap√© com termos de uso */}
        <div className="terms-text">
          <p>Ao fazer login, voc√™ concorda com nossos termos de uso</p>
        </div>
      </div>
    </div>
  );
};

export default LoginView;
